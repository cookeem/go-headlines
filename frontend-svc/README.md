# 如何安装运行

```sh
npm install webpack webpack-cli -D
npm install react react-dom -S
npm install @material-ui/core @material-ui/icons -S
npm install babel-loader@8 @babel/core @babel/preset-env @babel/preset-react @babel/plugin-syntax-dynamic-import -D
npm install css-loader style-loader file-loader -D
npm install html-webpack-plugin -D
npm install webpack-dev-server -D

# 运行调试
npm run start

# 正式打包
npm run build

# 打包输出目录
dist
```
